function doGet(e) { if (e && e.parameter && e.parameter.api) { const api = e.parameter.api; const routeName = e.parameter.routeName || ''; let result = []; if (api === 'busLocations') { result = getBusLocations(); } else if (api === 'routePath') { result = getRoutePath(routeName); } else if (api === 'restrictedPath') { result = getRestrictedPath(routeName); } else if (api === 'stops') { result = getStops(routeName); } else if (api === 'notices') { result = getNotices(); } else { result = { error: 'unknown api' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } const page = e && e.parameter && e.parameter.page ? e.parameter.page : 'map'; if (page === 'sender') { const template = HtmlService.createTemplateFromFile('sender'); template.busId = e && e.parameter && e.parameter.busId ? e.parameter.busId : 'BUS01'; template.routeName = e && e.parameter && e.parameter.routeName ? e.parameter.routeName : '鴨志田'; template.statusText = e && e.parameter && e.parameter.status ? e.parameter.status : '運行中'; template.autoStart = e && e.parameter && e.parameter.auto === '1'; return template.evaluate() .addMetaTag('charset', 'UTF-8') .addMetaTag('viewport', 'width=device-width, initial-scale=1, shrink-to-fit=no') .setTitle('送迎バス位置送信'); } return HtmlService.createHtmlOutputFromFile('map') .addMetaTag('charset', 'UTF-8') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setTitle('送迎バス現在地'); }
